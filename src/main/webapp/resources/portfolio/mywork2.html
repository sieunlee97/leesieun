    - 마우스 오버 효과 (work.html/work.jsp)

    
  /*마우스를 풀오버했을 때 나오는 배경색상, before가상요소 영역*/
  .work-section li a::before {
      display:block;
      width:100%;
      height:100%;
      position:absolute;  /*absolute 위치는 위 a태그에서 지정한 relative 위치에 한정해서 절대위치로 지정*/
      top:0px;            /*화면의 세로로 상단 0px 위치*/
      left:0px;           /*화면의 가로로 왼쪽 0px 위치*/
      background:#000;    /*배경색상 #000 (black)*/
      content:"";         /*content 없음*/
      opacity:0;          /*투명도 0%로 지정(안보임)*/
  }
  .work-section li a:hover::before {
      opacity:0.7;        /*투명도 0.7로 지정*/
  }
